cmake_minimum_required(VERSION 3.10)
project(TestUsuarios)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)  # Para asegurar que se use C++17

# Incluir directorios donde están tus headers
include_directories(
    ../../Core
    ../../DataStructures
    ../../Logic
)

# Archivos fuente que deseas compilar
set(SOURCES
    ProductoManager.cpp
    ../../DataStructures/ListaEnlazadaProductos.cpp
)

# Crear el ejecutable
add_executable(ProductoManager ${SOURCES})

# Añadir directorios de includes para el target (mejor usar target_include_directories)
target_include_directories(ProductoManager PRIVATE
    ../../Core
    ../../DataStructures
    ../../Logic  # Corrige la mayúscula aquí (antes pusiste "logic")
)
